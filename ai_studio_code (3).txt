"use client";
import React, { useEffect, useState } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { 
  Plus, Link as LinkIcon, Globe, MapPin, ShieldCheck, 
  Clock, Zap, CheckCircle2, Loader2, AlertCircle 
} from 'lucide-react';

export default function ClientDashboard() {
  const [nodes, setNodes] = useState(70);
  const [url, setUrl] = useState('');
  const [platform, setPlatform] = useState('instagram');
  const [provinces, setProvinces] = useState([]);
  const [selectedProvince, setSelectedProvince] = useState('');
  const [selectedRegency, setSelectedRegency] = useState('');
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [showSuccess, setShowSuccess] = useState(false);
  const [showToast, setShowToast] = useState(false);

  const PRICE_PER_NODE = 1500;
  const totalPrice = nodes * PRICE_PER_NODE;

  const handleOrderSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsSubmitting(true);

    // MENGGUNAKAN NEXT_PUBLIC AGAR BISA DIAKSES DI CLIENT SIDE
    const webhookUrl = process.env.NEXT_PUBLIC_N8N_WEBHOOK_URL;

    const payload = {
      client_id: "dummy-client-123",
      target_url: url,
      platform,
      nodes_quantity: nodes,
      province_id: selectedProvince,
      city_id: selectedRegency,
      total_price: totalPrice
    };

    try {
      const response = await fetch(webhookUrl || '', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });

      if (response.ok) {
        setShowToast(true);
        setShowSuccess(true);
      } else {
        throw new Error('Gagal');
      }
    } catch (error) {
      alert("Terjadi kesalahan koneksi ke server.");
    } finally {
      setIsSubmitting(false);
      setTimeout(() => setShowToast(false), 5000);
    }
  };

  return (
    <div className="pt-24 pb-16 px-4 max-w-7xl mx-auto">
      <h1 className="text-3xl font-bold text-white mb-8">Ruang Kendali Klien</h1>
      <form onSubmit={handleOrderSubmit} className="p-8 rounded-3xl bg-slate-900 border border-slate-800">
        <input 
          required type="url" value={url} onChange={(e) => setUrl(e.target.value)}
          className="w-full bg-slate-950 border border-slate-800 rounded-xl p-3 text-white mb-6"
          placeholder="URL Target..."
        />
        {/* ... (Gunakan sisa form dari ClientDashboard.tsx sebelumnya) ... */}
        <button 
          disabled={isSubmitting}
          className="w-full py-4 bg-neon-cyan text-slate-950 font-black rounded-xl disabled:opacity-50"
        >
          {isSubmitting ? <Loader2 className="animate-spin mx-auto" /> : "BAYAR & EKSEKUSI"}
        </button>
      </form>
      {/* ... (Success Modal & Toast) ... */}
    </div>
  );
}
